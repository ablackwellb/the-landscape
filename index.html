<script>

(function(){

  const reduceMotion = window.matchMedia('(prefers-reduced-motion: reduce)').matches;


  // ---------- Moments (samples until a real one begins) ----------

  let moments = [

    { year:1968, z:-18, depth: 0.35,  label:"A new beginning" },

    { year:1985, z: 22, depth:-0.85, label:"The hardest year" },

    { year:1998, z:-10, depth: 0.15,  label:"Ordinary, good" },

    { year:2008, z: 28, depth:-0.55, label:"A season of loss" },

    { year:2016, z:-24, depth: 0.9,   label:"The work that felt like purpose" },

    { year:2026, z: 10, depth: 0.6,   label:"Still climbing" }

  ];

  let usingSamples = true;


  // Map years onto the land, oldest at the far edge, newest nearest today

  function layoutMoments(){

    if(!moments.length) return;

    const years = moments.map(m=>m.year);

    const minY = Math.min(...years), maxY = Math.max(...years);

    if(maxY === minY){

      moments.forEach((m,i)=>{ m.t = -60 + i*40; });

    } else {

      moments.forEach(m=>{ m.t = -170 + (m.year-minY)/(maxY-minY)*340; });

    }

    // nudge same-year moments apart so both stay visible

    const sorted=[...moments].sort((a,b)=>a.t-b.t);

    for(let i=1;i<sorted.length;i++){

      if(sorted[i].t - sorted[i-1].t < 7) sorted[i].t = sorted[i-1].t + 7;

    }

  }


  const PEAK_AMP=58, VALLEY_AMP=52, SIGMA=34;

  function terrainHeight(x,z){

    let h = Math.sin(x*0.018)*Math.cos(z*0.022)*4.5

          + Math.sin(x*0.041+1.7)*Math.sin(z*0.037)*2.2;

    for(const m of moments){

      const dx=x-m.t, dz=z-m.z;

      const g=Math.exp(-(dx*dx+dz*dz)/(2*SIGMA*SIGMA));

      h += (m.depth>=0 ? m.depth*PEAK_AMP : m.depth*VALLEY_AMP)*g;

    }

    return h;

  }


  // ---------- Scene ----------

  const scene = new THREE.Scene();

  // A near-black indigo atmosphere lets the wire terrain carry the image.

  scene.background = new THREE.Color(0x080d1d);

  scene.fog = new THREE.Fog(0x0e1631, 330, 790);

  const camera = new THREE.PerspectiveCamera(46, innerWidth/innerHeight, 1, 2000);

  const renderer = new THREE.WebGLRenderer({antialias:true});

  renderer.setPixelRatio(Math.min(devicePixelRatio,2));

  renderer.setSize(innerWidth, innerHeight);

  renderer.outputEncoding = THREE.sRGBEncoding;

  document.getElementById('scene').appendChild(renderer.domElement);


  // Illuminated grid gives landscape unfinished look

  scene.add(new THREE.HemisphereLight(0x79e6df, 0x08091b, 0.72));

  const sun = new THREE.DirectionalLight(0xf2c17d, 0.78);

  sun.position.set(-220,150,160); scene.add(sun);

  const fill = new THREE.DirectionalLight(0x536bd1, 0.28);

  fill.position.set(180,90,-140); scene.add(fill);


  let terrainMesh=null, wireMesh=null;

  const pathGroup = new THREE.Group(); scene.add(pathGroup);

  const markerGroup = new THREE.Group(); scene.add(markerGroup);


  // A moderately dense mesh keeps each grid cell visible rather than merging to solid mass @ reg distances

  const W=460, D=300, SEG=150;


  // Elevation palette: deep indigo valleys to blue/violet to teal/cyan land then to gold crests.

  const cValleyDeep=new THREE.Color(0x111631);

  const cValley=new THREE.Color(0x283877);

  const cBase=new THREE.Color(0x168697);

  const cHigh=new THREE.Color(0x68ded5);

  const cCrest=new THREE.Color(0xfff399);


  function buildTerrain(){

    if(terrainMesh){

      scene.remove(terrainMesh);

      terrainMesh.geometry.dispose();

      terrainMesh.material.dispose();

    }

    if(wireMesh){

      scene.remove(wireMesh);

      wireMesh.geometry.dispose();

      wireMesh.material.dispose();

    }


    const geo = new THREE.PlaneGeometry(W, D, SEG, Math.round(SEG*D/W));

    geo.rotateX(-Math.PI/2);

    const pos = geo.attributes.position;

    const colors=[];


    for(let i=0;i<pos.count;i++){

      const x=pos.getX(i), z=pos.getZ(i);

      const h=terrainHeight(x,z);

      pos.setY(i,h);


      const c=new THREE.Color();

      if(h<0){

        // Valleys will shift through blue-violet into a nearly black indigo.

        const t=Math.min(1,-h/52);

        c.copy(cBase).lerp(cValley, Math.pow(t,0.72));

        if(t>0.62) c.lerp(cValleyDeep,(t-0.62)/0.38);

      }else{

        // Upper land brightens through cyan before the highest points turn gold

        const t=Math.min(1,h/58);

        c.copy(cBase).lerp(cHigh,Math.pow(t,0.82));

        if(t>0.64) c.lerp(cCrest,(t-0.64)/0.36);

      }

      colors.push(c.r,c.g,c.b);

    }


    geo.setAttribute('color', new THREE.Float32BufferAttribute(colors,3));

    geo.computeVertexNormals();


    // The ghosted surface gives the valleys weight without hiding the mesh.

    terrainMesh = new THREE.Mesh(geo, new THREE.MeshLambertMaterial({

      vertexColors:true,

      transparent:true,

      opacity:0.22,

      side:THREE.DoubleSide,

      depthWrite:true

    }));

    terrainMesh.renderOrder=1;

    scene.add(terrainMesh);


    // MeshBasicMaterial keeps the lines luminous even inside deep valleys.

    wireMesh = new THREE.Mesh(geo.clone(), new THREE.MeshBasicMaterial({

      vertexColors:true,

      wireframe:true,

      transparent:true,

      opacity:0.82,

      depthWrite:false

    }));

    wireMesh.position.y=0.32;

    wireMesh.renderOrder=2;

    scene.add(wireMesh);

  }


  // Path passes exactly through every moment — smooth in plan view,

  // hugging the terrain in height. It dips to meet every valley.

  function buildPath(){

    while(pathGroup.children.length){

      const c=pathGroup.children.pop(); c.geometry.dispose();

    }

    if(moments.length<2) return;

    const sorted=[...moments].sort((a,b)=>a.t-b.t);

    const ctrl=[];

    const first=sorted[0], last=sorted[sorted.length-1];

    ctrl.push(new THREE.Vector3(first.t-38, 0, first.z));

    for(const m of sorted) ctrl.push(new THREE.Vector3(m.t, 0, m.z));

    ctrl.push(new THREE.Vector3(last.t+38, 0, last.z));

    const planCurve=new THREE.CatmullRomCurve3(ctrl, false, 'centripetal');

    const pts=[];

    const N=300;

    for(let i=0;i<=N;i++){

      const p=planCurve.getPoint(i/N);

      pts.push(new THREE.Vector3(p.x, terrainHeight(p.x,p.z)+1.6, p.z));

    }

    const curve=new THREE.CatmullRomCurve3(pts);

    pathGroup.add(new THREE.Mesh(

      new THREE.TubeGeometry(curve,340,0.9,8,false),

      new THREE.MeshBasicMaterial({color:0xf2c17d})

    ));

    pathGroup.add(new THREE.Mesh(

      new THREE.TubeGeometry(curve,340,2.4,8,false),

      new THREE.MeshBasicMaterial({color:0xf2c17d,transparent:true,opacity:0.16,blending:THREE.AdditiveBlending,depthWrite:false})

    ));

  }


  function makeLabel(text){

    const pad=18, fs=30;

    const meas=document.createElement('canvas').getContext('2d');

    meas.font=`italic ${fs}px Georgia, serif`;

    const w=Math.ceil(meas.measureText(text).width)+pad*2;

    const c=document.createElement('canvas'); c.width=w; c.height=fs+pad*1.6;

    const g=c.getContext('2d');

    g.font=`italic ${fs}px Georgia, serif`;

    g.fillStyle='rgba(23,26,46,0.55)'; g.fillRect(0,0,c.width,c.height);

    g.fillStyle='#e8e6df'; g.textBaseline='middle';

    g.fillText(text,pad,c.height/2+1);

    const tex=new THREE.CanvasTexture(c); tex.minFilter=THREE.LinearFilter;

    const spr=new THREE.Sprite(new THREE.SpriteMaterial({map:tex,transparent:true,depthTest:false}));

    spr.scale.set(c.width*0.24, c.height*0.24, 1);

    return spr;

  }


  function buildMarkers(){

    while(markerGroup.children.length){

      const c=markerGroup.children.pop();

      if(c.geometry) c.geometry.dispose();

      if(c.material&&c.material.map) c.material.map.dispose();

    }

    const sorted=[...moments].sort((a,b)=>a.t-b.t);

    sorted.forEach((m,idx)=>{

      const y=terrainHeight(m.t,m.z), isPeak=m.depth>=0;

      const marker=new THREE.Mesh(

        new THREE.SphereGeometry(2.1,16,16),

        new THREE.MeshBasicMaterial({color:isPeak?0xf2c17d:0xaeb6e8})

      );

      marker.position.set(m.t,y+2.2,m.z); markerGroup.add(marker);

      const halo=new THREE.Mesh(

        new THREE.SphereGeometry(3.8,16,16),

        new THREE.MeshBasicMaterial({color:isPeak?0xf2c17d:0xaeb6e8,transparent:true,opacity:0.18,blending:THREE.AdditiveBlending,depthWrite:false})

      );

      halo.position.copy(marker.position); markerGroup.add(halo);

      const label=makeLabel(m.year + " — " + m.label);

      // stagger heights so neighbors don't stack

      label.position.set(m.t, y+(isPeak?16:13)+(idx%3)*4.5, m.z);

      label.userData.isLabel=true;

      label.userData.baseScale=label.scale.clone();

      markerGroup.add(label);

    });

  }


  function rebuildAll(){ layoutMoments(); buildTerrain(); buildPath(); buildMarkers(); }

  rebuildAll();


  // ---------- Composer ----------

  const $=id=>document.getElementById(id);

  let zAlt=1;


  $('add').addEventListener('click',()=>{

    const label=$('moment').value.trim();

    if(!label){ $('moment').focus(); return; }

    let year=parseInt($('year').value,10);

    if(isNaN(year)) year=new Date().getFullYear();

    year=Math.max(1900,Math.min(2100,year));

    if(usingSamples){ moments=[]; usingSamples=false; }

    zAlt*=-1;

    const z=zAlt*(12+Math.random()*20);

    moments.push({year,z,depth:$('depth').value/100,label});

    $('moment').value=''; $('year').value=''; $('depth').value=0;

    rebuildAll();

  });

  $('moment').addEventListener('keydown',e=>{ if(e.key==='Enter') $('add').click(); });

  $('year').addEventListener('keydown',e=>{ if(e.key==='Enter') $('add').click(); });

  $('clear').addEventListener('click',()=>{ moments=[]; usingSamples=false; rebuildAll(); });


  // Keep a copy — the file will belong to user on their device

  $('save').addEventListener('click',()=>{

    const data={version:1, kept:new Date().toISOString().slice(0,10),

                moments:moments.map(m=>({label:m.label,year:m.year,depth:m.depth,z:m.z}))};

    const blob=new Blob([JSON.stringify(data,null,2)],{type:'application/json'});

    const a=document.createElement('a');

    a.href=URL.createObjectURL(blob);

    a.download='my-landscape.json';

    a.click();

    URL.revokeObjectURL(a.href);

  });


  // Reopen a saved map

  $('load').addEventListener('click',()=>$('loadfile').click());

  $('loadfile').addEventListener('change',e=>{

    const f=e.target.files[0];

    if(!f) return;

    const r=new FileReader();

    r.onload=()=>{

      try{

        const data=JSON.parse(r.result);

        if(!Array.isArray(data.moments)) throw 0;

        moments=data.moments

          .filter(m=>m&&typeof m.label==='string'&&typeof m.depth==='number')

          .map(m=>({

            label:String(m.label).slice(0,140),

            year:Math.max(1900,Math.min(2100,parseInt(m.year,10)||new Date().getFullYear())),

            depth:Math.max(-1,Math.min(1,m.depth)),

            z:typeof m.z==='number'?m.z:(Math.random()>0.5?1:-1)*(12+Math.random()*20)

          }));

        usingSamples=false;

        rebuildAll();

      }catch(_){ /* not a landscape file — land unchanged */ }

      e.target.value='';

    };

    r.readAsText(f);

  });


  // Collapse toggle — the land deserves the whole screen sometimes

  const composer=$('composer'), mark=$('toggleMark');

  $('composerHead').addEventListener('click',()=>{

    composer.classList.toggle('collapsed');

    mark.textContent=composer.classList.contains('collapsed')?'+ add a moment':'— hide';

  });

  // start collapsed on small screens so the land is what greets you

  if(innerWidth<640){ composer.classList.add('collapsed'); mark.textContent='+ add a moment'; }


  // ---------- Camera ----------

  const target=new THREE.Vector3(0,0,0);

  let azim=-0.35, elev=0.72, dist=330;

  let dragging=false, px=0, py=0, idleSpin=!reduceMotion;

  function placeCamera(){

    elev=Math.max(0.12,Math.min(1.35,elev));

    dist=Math.max(120,Math.min(620,dist));

    camera.position.set(

      target.x+dist*Math.cos(elev)*Math.sin(azim),

      target.y+dist*Math.sin(elev),

      target.z+dist*Math.cos(elev)*Math.cos(azim)

    );

    camera.lookAt(target);

  }

  const el=renderer.domElement;

  el.addEventListener('pointerdown',e=>{dragging=true;idleSpin=false;px=e.clientX;py=e.clientY;});

  addEventListener('pointerup',()=>dragging=false);

  addEventListener('pointermove',e=>{

    if(!dragging)return;

    azim-=(e.clientX-px)*0.0042;

    elev+=(e.clientY-py)*0.0032;

    px=e.clientX;py=e.clientY;

  });

  el.addEventListener('wheel',e=>{e.preventDefault();idleSpin=false;dist+=e.deltaY*0.35;},{passive:false});

  addEventListener('resize',()=>{

    camera.aspect=innerWidth/innerHeight;

    camera.updateProjectionMatrix();

    renderer.setSize(innerWidth,innerHeight);

  });


  const NEAR=320, FAR=580;

  function updateLabels(){

    for(const c of markerGroup.children){

      if(!c.userData.isLabel) continue;

      const d=camera.position.distanceTo(c.position);

      let o=1-(d-NEAR)/(FAR-NEAR);

      o=Math.max(0,Math.min(1,o));

      c.material.opacity=0.25+0.75*o*o;

      c.visible=d<FAR+60;

      const s=0.85+0.15*o;

      c.scale.set(c.userData.baseScale.x*s, c.userData.baseScale.y*s, 1);

    }

  }


  function animate(){

    requestAnimationFrame(animate);

    if(idleSpin) azim+=0.00055;

    placeCamera();

    updateLabels();

    renderer.render(scene,camera);

  }

  placeCamera(); animate();

})();
